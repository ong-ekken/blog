for guessesTaken in range(1, 7):
    print('Take a guess.')
    guess = int(input())

    if guess < secretNumber:
        print('Your guess is too low.')
    elif guess > secretNumber:
        print('Your guess is too high.')
    else:
        break # This condition is the correct guess!

if guess == secretNumber:
    print('Good job! You guessed my number in ' + str(guessesTaken) + ' guesses!')

    print('Nope. The number I was thinking of was ' + str(secretNumber))

Project: Rock Paper Scissors 

import random, sys

print('ROCK, PAPER, SCISSORS')

# These variables keep track of the number of wins, losses, and ties.
wins = 0
losses = 0
ties = 0

while True: # The main game loop.
    print('%s Wins, %s Losses, %s Ties' % (wins, losses, ties))
    while True: # The player input loop.
        print('Enter your move: (r)ock (p)aper (s)cissors or (q)uit')
        playerMove = input()
        if playerMove == 'q':
            sys.exit() # Quit the program.
        if playerMove == 'r' or playerMove == 'p' or playerMove == 's':
            break # Break out of the player input loop.
        print('Type one of r, p, s, or q.')

    # Display what the player chose:
    if playerMove == 'r':
        print('ROCK versus...')
    elif playerMove == 'p':
        print('PAPER versus...')
    elif playerMove == 's':
        print('SCISSORS versus...')

    # Display what the computer chose:
    randomNumber = random.randint(1, 3)
    if randomNumber == 1:
        computerMove = 'r'
        print('ROCK')
    elif randomNumber == 2:
        computerMove = 'p'
        print('PAPER')
    elif randomNumber == 3:
        computerMove = 's'
        print('SCISSORS')

    # Display and record the win/loss/tie:
    if playerMove == computerMove:
        print('It is a tie!')
        ties = ties + 1
    elif playerMove == 'r' and computerMove == 's':
        print('You win!')
        wins = wins + 1
    elif playerMove == 'p' and computerMove == 'r':
        print('You win!')
        wins = wins + 1
    elif playerMove == 's' and computerMove == 'p':
        print('You win!')
        wins = wins + 1
    elif playerMove == 'r' and computerMove == 'p':
        print('You lose!')
        losses = losses + 1
    elif playerMove == 'p' and computerMove == 's':
        print('You lose!')
        losses = losses + 1
    elif playerMove == 's' and computerMove == 'r':
        print('You lose!')
        losses = losses + 1

Practice Questions

  1. What are the two values of the Boolean data type? How do you write them?

    • True
    • False
    • initial-capitalized

  2. What are the three Boolean operators?

    • and
    • or
    • not

  3. Write out the truth tables of each Boolean operator (that is, every possible combination of Boolean values for the operator and what they evaluate to).

    • True and True -> True
    • True and False -> False
    • False and False -> False
    • True or True -> True
    • True or False -> True
    • False or False -> False
    • not True -> False
    • not False -> True

  4. What do the following expressions evaluate to?

    • False : (5 > 4) and (3 == 5)
    • False : not (5 > 4)
    • True : (5 > 4) or (3 == 5)
    • False : not ((5 > 4) or (3 == 5))
    • False : (True and True) and (True == False)
    • True : (not False) or (not True)

  5. What are the six comparison operators?

    • more than

    • = more than or equals

    • < less than
    • <= less than or equals
    • == equals
    • != not equals

  6. What is the difference between the equal to operator and the assignment operator?

    • equal to operator is double equal signs ==
    • assignment operator is single equal sign =

  7. Explain what a condition is and where you would use one.

    • a condition evaluates to True or False
    • used to qualify whether to execute or to terminate a loop

  8. Identify the three blocks in this code:

    spam = 0 if spam == 10: -- Block 1

     print('eggs')
 if spam > 5:  -- Block 2
     print('bacon')  
 else:         -- Block 3
     print('ham')
 print('spam')

    print('spam')

  9. Write code that prints Hello if 1 is stored in spam, prints Howdy if 2 is stored in spam, and prints Greetings! if anything else is stored in spam.

    • if spam == 1: print("Hello")
    • elif spam == 2: print("Howdy")
    • else: print("Greetings!")

  10. What keys can you press if your program is stuck in an infinite loop?

    • Ctrl-C

  11. What is the difference between break and continue?

    • break exits the loop
    • continue starts the next iteration of the loop

  12. What is the difference between range(10), range(0, 10), and range(0, 10, 1) in a for loop?

    • no difference

  13. Write a short program that prints the numbers 1 to 10 using a for loop. Then write an equivalent program that prints the numbers 1 to 10 using a while loop.

    • for n in range(1,11): print(n)
    • m = 1
    • while m <= 10: print(m)

  14. If you had a function named bacon() inside a module named spam, how would you call it after importing spam?

    • spam.bacon()
