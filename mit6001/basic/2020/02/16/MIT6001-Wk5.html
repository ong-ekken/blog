Exercise Hand (only the definition of update method) 

 def update(self, word):
        """
        Does not assume that self.hand has all the letters in word.

        Updates the hand: if self.hand does have all the letters to make
        the word, modifies self.hand by using up the letters in the given word.

        Returns True if the word was able to be made with the letter in
        the hand; False otherwise.
        
        word: string
        returns: Boolean (if the word was or was not made)
        """
        # Your code here
        # raise NotImplementedError()
        temp = self.hand.copy()
        for i in word:
            if temp.get(i, 0):
                temp[i] -= 1
            else:    
                return False
        self.hand = temp
        return True

Exercise genPrimes 

def genPrimes():
    primes = []
    num = 2         #initialised with 2
    
    # Run forever
    while True:

        # Reset flag
        flag = True
        
        # If num is divisible by any prime,
        for prime in primes:
            if num % prime == 0:
                flag = False
        
        # Add to list of primes and yield value
        if flag:
            primes.append(num)
            yield num
        
        # Mechanism to consider only odd numbers
        if num % 2 == 1:
            num += 2
        else:
            num += 1

def keysWithValue(aDict, target):
    '''
    aDict: a dictionary
    target: an integer

    returns: list of keys sorted in increasing order that point to value of target
    '''
    # Your code here  
    output = []
    for keys in aDict:
        if aDict[keys] == target:
            output.append(keys)
    return sorted(output)

def sumDigits(N):
    '''
    N: a non-negative integer

    returns: sum of the digits in N
    '''
    # Your code here
    assert type(N) == int and N >= 0, "N is not a non-negative integer"
    if N < 10:
        return N
    else:
        return sumDigits(N // 10) + N % 10

def general_poly (L):
    """ L, a list of numbers (n0, n1, n2, ... nk)

    n0 * x^k + n1 * x^(k-1) + ... nk * x^0 """
